function draw(canvasId, text) { let names = text.split('،').filter(n => n.trim() !== ""); let canvas = document.getElementById(canvasId); let ctx = canvas.getContext("2d"); ctx.clearRect(0, 0, 250, 250); if (names.length === 0) return; let angle = (2 * Math.PI) / names.length; names.forEach((name, i) => { ctx.beginPath(); ctx.moveTo(125, 125); ctx.arc(125, 125, 120, i * angle, (i + 1) * angle); ctx.fillStyle = i % 2 === 0 ? "#795548" : "#5d4037"; ctx.fill(); ctx.stroke(); ctx.save(); ctx.translate(125, 125); ctx.rotate(i * angle + angle / 2); ctx.fillStyle = "white"; ctx.font = "16px Arial"; ctx.fillText(name, 50, 5); ctx.restore(); }); } // استرجاع البيانات عند التشغيل window.onload = () => { ['names1', 'names2'].forEach((k, i) => { let saved = localStorage.getItem(k); if(saved) { document.getElementById('in' + (i+1)).value = saved; draw('canvas' + (i+1), saved); } }); };
